(function ($) {
	"use strict";

	$.fn.gsaautocomplete = function (opts) {
		var options = this.extend({
			"url": "/suggest.php",
			"ulId": "suggest-results"
		}, opts);

		/**
		 * Plugin Begin
		 * turn plugin on all matched nodes
		 */
		return this.each(function () {

			/**
			 * globals
			 */
			var $this, $list, $keys, $ikeys, $i, $v;
			$this = $(this);
			$v = $this.val();
			$list = $("#" + options.ulId);
			$keys = [27, 38, 40];
			$i = 0;

			/**
			 * core
			 */
			$this.attr({
				"autocomplete": "off",
				"aria-autocomplete": "list",
				"aria-live": "polite",
				"aria-controls": options.ulId
			}).on({
				keyup: function (e) {

					if ($.inArray(e.keyCode, $keys) === -1) {
						if ($v !== $this.val()) {

							$.post(options.url, {

								"q": $this.val()

							}, function (data) {

								$list.remove();

								if (data.results.length > 0) {
									var items = [];

									for (var i in data.results) {
										items.push('<li role="listitem">' + data.results[i].name + '</li>');
									}

									$this.after(
									$("<ul>").attr({
										"id": options.ulId,
										"role": "list"
									}).html(items.join("")));
									$list = $("#" + options.ulId);
								}

								$list.on("click", "li", function () {
									$this.val($(this).text());
									$list.remove();
								});

								$("*").not("form *").on("click", function () {
									$list.remove();
								});

							}, "json"); //post

						}
					}

					if ($.inArray(e.keyCode, $keys) !== -1) {
						var len, lis;
						len = $list.children().length;
						lis = $list.children();

						switch (e.keyCode) {
						case 27:
							// KEY_ESC
							$list.remove();
							$i = -1;
							break;
						case 38:
							// KEY_UP
							if ($i === 0 || $i === -1) {
								$i = len - 1;
							} else {
								$i--;
							}
							break;
						case 40:
							// KEY_DOWN
							if ($i === len - 1) {
								$i = 0;
							} else {
								$i++;
							}
							break;
						default:
							break;
						}
						lis.each(function (i) {
							if (i === $i) {
								$(this).addClass("active");
								$this.val(this.textContent);
							} else {
								$(this).removeClass("active");
							}
						});
					} else {
						$i = -1;
					}
					$v = $this.val();
				} // keyup
			}); // $this chain
		}); //this.each
	}; // $.fn
})(jQuery); //$closure

require_once 'Zend.php';
require_once 'Zend/Uri/Exception.php';
require_once 'Zend/Uri/Http.php';
require_once 'Zend/Uri/Mailto.php';

abstract class Zend_Uri
{

	/**
	* Return a string representation of this URI.
	*
	* @see     getUri()
	* @return  string
	*/
	public function __toString()
	{
		return $this->getUri();
	}

	static public function factory($uri = 'http')
	{
		$uri = explode(':', $uri, 2);
		$scheme = strtolower($uri[0]);
		$schemeSpecific = isset($uri[1]) ? $uri[1] : '';

		// Security check: $scheme is used to load a class file,
		// so only alphanumerics are allowed.
		if (!ctype_alnum($scheme)) {
			throw new Zend_Uri_Exception('Illegal scheme');
		}
	}
}
